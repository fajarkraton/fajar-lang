@kernel
fn handle_timer_irq() {
    let ticks = port_read(0x40)
    scheduler.tick(ticks)
    irq_acknowledge(0)
}

@kernel
fn kmain() {
    irq_register(0, handle_timer_irq)
    irq_enable()
}

        to_g(raw[0], raw[1]),
        to_g(raw[2], raw[3]),
        to_g(raw[4], raw[5]),
    ]
}
enum Action { Hover, Land, Avoid(f64) }

fn decide(altitude: f64, obstacle: bool) -> Action {
    match (altitude, obstacle) {
        (h, _) if h < 1.0 => Action::Land,
        (_, true)           => Action::Avoid(2.0),
        _                    => Action::Hover,
    }
}

Built for the Hardware Edge

🧠

Embedded ML

First-class tensors, autograd, and quantized inference (FP4/FP8/INT8). Train and deploy models on Cortex-M, RISC-V, and Jetson Thor.

⚙️

OS Integration

Write interrupt handlers, page tables, and device drivers in the same language as your neural networks. Context annotations enforce isolation.

🛡️

Safety by Construction

Ownership-lite (no lifetime annotations), borrow checking, null safety via Option<T>, and exhaustive pattern matching. If it compiles, it's safe.

⚡

Native Performance

Cranelift and LLVM backends. AVX-512, AMX tile ops, CUDA Tensor Cores. Zero-cost abstractions from board-level HAL to GPU compute.
